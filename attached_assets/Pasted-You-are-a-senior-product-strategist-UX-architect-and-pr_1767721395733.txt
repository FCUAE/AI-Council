const StatsCounter = React.memo(({ end, suffix = "" }) => {
  const [count, setCount] = useState(0);
  const [isVisible, setIsVisible] = useState(false);
  const ref = useRef(null);

  useEffect(() => {
    const observer = new IntersectionObserver(([entry]) => {
      if (entry.isIntersecting) setIsVisible(true);
    });

    if (ref.current) observer.observe(ref.current);
    return () => observer.disconnect();
  }, []);

  useEffect(() => {
    if (!isVisible) return;

    const duration = 2000;
    const increment = end / (duration / 16);
    let current = 0;

    const timer = setInterval(() => {
      current += increment;
      if (current >= end) {
        setCount(end);
        clearInterval(timer);
      } else {
        setCount(Math.floor(current));
      }
    }, 16);

    return () => clearInterval(timer);
  }, [isVisible, end]);

  return (
    <div
      ref={ref}
      className="font-header text-4xl md:text-6xl text-white/95 mb-2"
    >
      <span className="text-[#39FF14]">
        {count}
        {suffix}
      </span>
    </div>
  );
});

const ReelCard = React.memo(({ reel }) => {
  const videoRef = useRef(null);
  const [isPlaying, setIsPlaying] = useState(false);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState(false);
  const [isHovering, setIsHovering] = useState(false);

  useEffect(() => {
    if (isHovering && videoRef.current && !isPlaying && !error) {
      videoRef.current.play().catch(() => {});
    } else if (!isHovering && videoRef.current && !isPlaying && !error) {
      videoRef.current.pause();
      videoRef.current.currentTime = 0;
    }
  }, [isHovering, isPlaying, error]);

  const togglePlay = (e) => {
    e.stopPropagation();
    if (videoRef.current && !error) {
      if (isPlaying) {
        videoRef.current.pause();
      } else {
        const playPromise = videoRef.current.play();
        if (playPromise !== undefined) {
          playPromise.catch((error) => {
            console.error("Playback failed:", error);
            setIsPlaying(false);
          });
        }
      }
      setIsPlaying(!isPlaying);
    }
  };

  const handleLoadedData = () => setIsLoading(false);

  const handleError = () => {
    console.error(`Error loading video for ${reel.id}`);
    setIsLoading(false);
    setError(true);
  };

  return (
    <div
      className="relative aspect-[9/16] w-full h-auto max-w-[320px] mx-auto bg-black/5 border border-black/10 rounded-2xl overflow-hidden group cursor-pointer transition-all duration-300 ease-out hover:border-black/20 shadow-[0_6px_18px_rgba(0,0,0,0.12)] hover:shadow-[0_10px_28px_rgba(0,0,0,0.16)] hover:scale-[1.02]"
      onClick={togglePlay}
      onMouseEnter={() => setIsHovering(true)}
      onMouseLeave={() => setIsHovering(false)}
      role="button"
      aria-label={`Play reel from ${reel.founderName} about ${reel.topic}`}
    >
      {isLoading && !error && (
        <div className="absolute inset-0 z-20 flex items-center justify-center bg-gray-50 scanlines">
          <div className="w-8 h-8 border-[3px] border-black/10 border-t-[#39FF14] rounded-full animate-spin"></div>
        </div>
      )}
      {error && (
        <div className="absolute inset-0 z-20 flex items-center justify-center bg-gray-50">
          <span className="text-xs font-bold text-black/50 uppercase font-pixel">
            Video Unavailable
          </span>
        </div>
      )}
      <video
        ref={videoRef}
        src={reel.videoUrl}
        poster={reel.thumbnail}
        className={`w-full h-full object-cover ${error ? "hidden" : ""}`}
        playsInline
        loop
        muted={!isPlaying}
        onLoadedData={handleLoadedData}
        onError={handleError}
      />
      {!error && (

const ApplicationForm = ({ isOpen, onClose, selectedPackage }) => {
  const [step, setStep] = useState(1);
  const [formStep, setFormStep] = useState(null);
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    company: "",
    monthlyBudget: "",
    startTimeline: "",
    currentStatus: "",
  });
  const [errors, setErrors] = useState({});
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [spotsLeft] = useState(() => {
    const today = new Date().getDate();
    return Math.max(1, 3 - (today % 3));
  });
  const [timeLeft, setTimeLeft] = useState("");

  // Lock body scroll when form is open to prevent background processing and interaction conflicts
  useEffect(() => {
    if (isOpen) {
      document.body.classList.add("modal-open");
    } else {
      document.body.classList.remove("modal-open");
    }
    return () => document.body.classList.remove("modal-open");
  }, [isOpen]);

  useEffect(() => {
    const endOfMonth = new Date();
    endOfMonth.setMonth(endOfMonth.getMonth() + 1);
    endOfMonth.setDate(0);
    endOfMonth.setHours(23, 59, 59);

    const updateTime = () => {
      const now = new Date();
      const diff = endOfMonth - now;
      const days = Math.floor(diff / (1000 * 60 * 60 * 24));
      setTimeLeft(`${days} days left`);
    };

    updateTime();
    const timer = setInterval(updateTime, 1000 * 60 * 60);

    return () => clearInterval(timer);
  }, []);

  const validateEmail = (email) => {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (!validateEmail(formData.email)) {
      setErrors({ ...errors, email: "Please enter a valid email" });
      return;
    }

    setIsSubmitting(true);
    await new Promise((resolve) => setTimeout(resolve, 1500));
    setIsSubmitting(false);
    setStep(2);
  };

  const handleClose = () => {
    setStep(1);
    setFormStep(null);
    setErrors({});
    setFormData({
      name: "",
      email: "",
      company: "",
      monthlyBudget: "",
      startTimeline: "",
      currentStatus: "",
    });
    onClose();
  };

  if (!isOpen) return null;

  if (formStep === "quiz") {
    return (
      <div className="fixed inset-0 z-[10000] flex items-center justify-center p-4 bg-black/60 backdrop-blur-md">
        <div className="relative w-full max-w-lg bg-white border-[2px] border-black/80 rounded-2xl shadow-[4px_4px_16px_rgba(0,0,0,0.12)] p-8 animate-fadeIn">
          <button
            onClick={() => setFormStep(null)}
            className="absolute top-4 right-4 text-black/60 hover:text-[#39FF14] transition-colors"
          >
            <X size={24} />
          </button>

          <h3 className="font-header text-2xl uppercase text-black/90 mb-6 text-center">
            Which describes you best?
          </h3>
          <div className="space-y-4">
            {[
              "Pre-launch / Building MVP",
              "Launched / Early traction",
              "Scaling / Growth Stage",
            ].map((status) => (
              <button
                key={status}
                onClick={() => {
                  setFormData({ ...formData, currentStatus: status });
                  setFormStep(null);
                }}
                className="w-full text-left p-4 border border-black/10 rounded-xl hover:border-black/80 hover:bg-[#39FF14] hover:shadow-md font-bold transition-all duration-200 text-sm uppercase text-black/80"
              >
                {status}
              </button>
            ))}
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className="fixed inset-0 z-[10000] flex items-center justify-center p-4 bg-black/60 backdrop-blur-md">
      <div className="relative w-full max-w-2xl bg-white border border-black/80 rounded-3xl shadow-[0_20px_40px_rgba(0,0,0,0.2)] max-h-[90vh] overflow-y-auto custom-scrollbar animate-fadeIn">
        <button
          onClick={handleClose}
          aria-label="Close form"
          className="absolute top-4 right-4 w-10 h-10 bg-black/5 border border-transparent hover:border-black/10 text-black/60 rounded-full flex items-center justify-center hover:bg-[#39FF14] hover:text-black transition-all duration-200 z-10"
        >
          <X size={20} strokeWidth={2.5} />
        </button>

        {step === 1 ? (
          <div className="p-6 md:p-12">
            <div className="mb-6">
              <div className="flex items-center justify-between mb-4">
                <div className="inline-block bg-[#39FF14] px-4 py-1 text-[10px] font-black uppercase tracking-widest rounded-full border border-black/5">
                  Step 1 of 2
                </div>
              </div>

              <div className="w-full h-2 bg-black/5 rounded-full mb-6 overflow-hidden">
                <div
                  className="h-full bg-[#39FF14] rounded-full transition-all duration-300 ease-out"
                  style={{ width: step === 1 ? "50%" : "100%" }}
                />
              </div>

              <h2 className="font-header text-3xl md:text-4xl uppercase text-black/90 mb-3">

                  onClick={() => openFolder("reels")}
                  className="group px-6 py-3 sm:px-8 sm:py-4 bg-white text-black font-black text-sm sm:text-base uppercase border-2 border-black/90 rounded-xl hover:bg-[#39FF14] hover:text-black hover:border-transparent transition-all shadow-[4px_4px_16px_rgba(0,0,0,0.12)] hover:shadow-[6px_6px_20px_rgba(0,0,0,0.15)] hover:scale-[1.02]"
                >
                  <span className="relative z-10 flex items-center gap-2">
                    View Reels
                    <ArrowRight
                      size={18}
                      className="group-hover:translate-x-1 transition-transform"
                      strokeWidth={2.5}
                    />
                  </span>
                </button>
              </div>
            </div>

            <div className="absolute inset-0 z-10 pointer-events-none">
              {isMobile ? (
                // Use a grid for better tablet optimization (2 cols for tablets, 1 for mobile)
                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4 p-6 pt-[380px] w-full max-w-4xl mx-auto">
                  {folders.map((folder) => (
                    <div
                      key={folder.id}
                      className="w-full pointer-events-auto"
                      onClick={() => setActiveView(folder.id)}
                    >
                      <div className="bg-white border border-black/5 p-5 flex items-center gap-4 rounded-xl shadow-[4px_4px_16px_rgba(0,0,0,0.08)] hover:shadow-[6px_6px_20px_rgba(0,0,0,0.12)] hover:scale-[1.02] active:scale-[0.98] transition-all cursor-pointer">
                        <div className="w-14 h-14 bg-gradient-to-br from-blue-400 to-blue-600 border border-blue-200/20 rounded-lg flex items-center justify-center flex-shrink-0 shadow-inner">
                          <folder.icon
                            size={24}
                            className="text-white"
                            strokeWidth={2.5}
                          />
                        </div>
                        <div className="flex-1 min-w-0">
                          <div className="text-xs font-black uppercase tracking-wider text-black/80 mb-1">
                            {folder.badge}
                          </div>
                          <div className="text-[10px] font-bold text-black/60 font-mono truncate">
                            {folder.name}
                          </div>
                        </div>
                        <div className="w-10 h-10 border border-black/10 rounded-full flex items-center justify-center flex-shrink-0 bg-[#39FF14]/80 shadow-sm">
                          <ArrowRight
                            size={18}
                            className="text-black"
                            strokeWidth={2.5}
                          />
                        </div>
                      </div>
                    </div>
                  ))}
                </div>
              ) : (
                folders.map((folder) => (
                  <div
                    key={folder.id}
                    className={`${
                      draggingId === folder.id ? "folder-drag-active" : ""
                    }`}
                  >
                    <Folder
                      folder={folder}
                      position={folder.initialPosition}
                      onMouseDown={handleDragStart}
                      onClick={() => handleFolderClick(folder.id)}
                    />
                  </div>
                ))
              )}
            </div>
          </div>

          <div className="md:hidden fixed bottom-0 left-0 right-0 z-50 bg-white/95 backdrop-blur-md border-t border-black/5 p-4 safe-area-bottom shadow-[0_-4px_20px_rgba(0,0,0,0.05)]">
            <button
              onClick={() => handleOpenForm("growth")}
              className="w-full bg-[#39FF14] text-black font-extrabold uppercase py-4 rounded-xl border border-transparent shadow-[0_4px_12px_rgba(57,255,20,0.3)] active:scale-[0.98] transition-all flex items-center justify-center gap-2"
            >
              <span>Book Free Strategy Call</span>
              <ArrowRight size={20} strokeWidth={2.5} />
            </button>
            <div className="flex items-center justify-center gap-4 mt-3">
              <p className="text-center text-[10px] text-black/40 uppercase tracking-wider">
                2 spots left this month • 15 min call
              </p>
              <button
                onClick={() => setActiveView("about")}
                className="text-[10px] text-black/60 uppercase tracking-wider font-bold hover:text-[#39FF14] transition-colors"
              >
                About Us →
              </button>
            </div>
          </div>

          {showFloatingCTA && (
            <div className="md:hidden fixed bottom-4 right-4 z-[60] animate-[fadeIn_0.3s_ease-out]">
              <button
                onClick={() => handleOpenForm("growth")}
                className="w-16 h-16 bg-[#39FF14] border border-black/5 rounded-full flex items-center justify-center shadow-[0_8px_24px_rgba(57,255,20,0.4)] active:scale-[0.95] transition-all"
              >
                <ArrowRight
                  size={24}
                  strokeWidth={2.5}
                  className="text-black"
                />
              </button>
            </div>
          )}
        </>
      )}

      {activeView === "websites" && (
        <FullPageView
          isOpen={true}
          folder={websitesFolder}
          onClose={() => setActiveView("desktop")}
          onOpenForm={handleOpenForm}
          setActiveView={setActiveView}
        />
      )}

      {activeView === "reels" && (
        <FullPageView
          isOpen={true}
          folder={reelsFolder}
          onClose={() => setActiveView("desktop")}
          onOpenForm={handleOpenForm}
          setActiveView={setActiveView}
        />
      )}

      {activeView === "about" && (
        <FullPageView
          isOpen={true}
          folder={aboutFolder}
          onClose={() => setActiveView("desktop")}
          onOpenForm={handleOpenForm}
          setActiveView={setActiveView}
        />
      )}

      <ApplicationForm
        isOpen={formOpen}
        onClose={() => {
          setFormOpen(false);
          setSelectedPackage(null);
        }}
        selectedPackage={selectedPackage}
      />
    </div>
  );
}
